#include <stdio.h>

int main() {
    int duracao, horas, minutos, segundos;

    //scanf("%d", &duracao);
    duracao = 556;

    horas = duracao / 3600;
    minutos = (duracao - horas * 3600) / 60;
    segundos = (duracao - ((horas * 3600) + (minutos * 60)));

    printf("%d:%d:%d\n", horas, minutos, segundos);

    return 0;
}
